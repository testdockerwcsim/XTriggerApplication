178 {
179  double x, y, r, z, xdir, ydir;
180  const int maxcount = 100;
181 
182  //create a flat distribution in r and phi
183  if(fFlatR
184  && (fXDistribution == kUniform)
185  && (fYDistribution == kUniform)) {
186  r = fRand->Uniform(0, +fMaxRPos);
187  fRand->Circle(xdir, ydir, 1);
188  x = r * xdir;
189  y = r * ydir;
190  }
191  //create a flat distribution in x and y
192  else {
193  r = fMaxRPos + 1;
194  while(r > fMaxRPos) {
195  //x

197  //y
198  y = GetRandomNumber(fYDistribution, fMaxRPos, fYMean, fYWidth, maxcount);
199  //r
200  r = TMath::Sqrt(TMath::Power(x, 2) + TMath::Power(y, 2));
201  }
202  }
203  //z
204  z = GetRandomNumber(fZDistribution, fMaxZPos, fZMean, fZWidth, maxcount);
205 
206  pos[0] = x;
