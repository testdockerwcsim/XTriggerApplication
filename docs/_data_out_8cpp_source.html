312  new_trig->SetNumTubesHit(old_trig->GetNumTubesHit());
313  WCSimRootCherenkovHit * hit;
314  WCSimRootCherenkovHitTime * hit_time;
315  for(int ihit = 0; ihit < old_trig->GetNcherenkovhits(); ihit++) {
316  TObject * obj = old_trig->GetCherenkovHits()->At(ihit);
317  if(!obj) continue;
318  hit = dynamic_cast<WCSimRootCherenkovHit*>(obj);
319  int tube_id = hit->GetTubeID();
320  std::vector<double> true_times;
321  std::vector<int> primary_parent_id;
322  for(int itime = hit->GetTotalPe(0); itime < hit->GetTotalPe(0) + hit->GetTotalPe(1); itime++) {
323  TObject * obj = old_trig->GetCherenkovHitTimes()->At(itime);
324  if(!obj) continue;

329  new_trig->AddCherenkovHit(tube_id, true_times, primary_parent_id);

331

333  WCSimRootTrack * track;
334  for(int itrack = 0; itrack < old_trig->GetNtrack_slots(); itrack++) {
335  TObject * obj = old_trig->GetTracks()->At(itrack);
336  if(!obj) continue;
337  track = dynamic_cast<WCSimRootTrack*>(obj);
338  new_trig->AddTrack(track);
339  }//itrack
340 
341  //set true digit parent info
342  // This is messy, since the WCSim digit order
343  // is different to the TriggerApplication digit order
344  //we're looping over triggers here
345  for(int itrigger = 0; itrigger < m_all_triggers->m_num_triggers; itrigger++) {
346  if(itrigger)
347  new_trig = wcsim_event->GetTrigger(itrigger);
348  double this_trigger_time = new_trig->GetHeader()->GetDate();
349  WCSimRootCherenkovDigiHit * new_digit;
350  WCSimRootCherenkovDigiHit * old_digit;
351  //not looping over "slots", because we just wrote this file, so we know there are no gaps
352  for(int idigit = 0; idigit < new_trig->GetNcherenkovdigihits(); idigit++) {
353  TObject * obj = new_trig->GetCherenkovDigiHits()->At(idigit);
354  if(!obj) continue;
355  new_digit = dynamic_cast<WCSimRootCherenkovDigiHit*>(obj);
356  int tube_id = new_digit->GetTubeId();
357  //get the time
358  double time = new_digit->GetT();
359  //and convert it back to the original (i.e. relative to WCSim's Date(), rather than
360  // TriggerApp's individual triggers' Date()s)
361  // - time_shift subtracts the WCSim Date, and subtracts the user-defined "offset"
362  // + this_trigger_time because hits are defined relative to their trigger time
363  time += - (time_shift / TimeDelta::ns) + this_trigger_time;
364 
365  //Now we find the new digit
366  bool found = false;
367  for(int idigit_old = 0; idigit_old < old_trig->GetNcherenkovdigihits_slots(); idigit_old++) {
368  TObject * obj = old_trig->GetCherenkovDigiHits()->At(idigit_old);

371  // First check tube id

373  // Then the time. Assume if we're within 0.5 ns it's the same hit

375  found = true;
376  break;
377  }//idigit_old

379  new_digit->SetPhotonIds(old_digit->GetPhotonIds());
380  }
381  }//idigit
382  }//itrigger
