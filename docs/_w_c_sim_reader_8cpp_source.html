436 SubSample WCSimReader::GetDigits()
437 {
438  // Store times relative to first hit time
439  TimeDelta first_time;
440  //loop over the digits
441  std::vector<int> PMTid;
442  std::vector<float> charge;
443  std::vector<TimeDelta::short_time_t> time;
444  for(int idigi = 0; idigi < m_wcsim_trigger->GetNcherenkovdigihits(); idigi++) {
445  //get a digit
446  TObject *element = (m_wcsim_trigger->GetCherenkovDigiHits())->At(idigi);
447  WCSimRootCherenkovDigiHit *digit =
448  dynamic_cast<WCSimRootCherenkovDigiHit*>(element);
449  //get the digit information
450  int ID = digit->GetTubeId();
451  if (idigi == 0){
452  // Store times relative to the first digit
453  first_time = TimeDelta(digit->GetT());
454  m_ss << "DEBUG: 1st digit time before shifting: " << first_time;
455  StreamToLog(DEBUG1);
456  }
457  TimeDelta::short_time_t T = (TimeDelta(digit->GetT()) - first_time) / TimeDelta::ns;
458  float Q = digit->GetQ();
459  PMTid.push_back(ID);
460  time.push_back(T);
461  charge.push_back(Q);
462  //print
463  if(idigi < 10 || m_verbose >= DEBUG2) {
464  m_ss << "DEBUG: Digit " << idigi
465  << " has T " << T
466  << ", Q " << Q
467  << " on PMT " << ID;
468  StreamToLog(DEBUG1);
469  }
470  }//idigi
471  m_ss << "DEBUG: Saved information on " << time.size() << " digits";
472  StreamToLog(DEBUG1);
473 
474  // Set the timestamp of the SubSample to the first digit time (the one every
475  // other time is relative to) plus the trigger time as reported by WCSim.
476  // WCSim stores all digit times relative to the trigger time.
477  //
478  // WCSim also adds a 950 ns offset to the digit times, if it no running in
479  // the NoTrigger mode. But we should not care about that here.
480  TimeDelta timestamp = TimeDelta(m_wcsim_trigger->GetHeader()->GetDate()) + first_time;
481  SubSample sub;
482  if (not sub.Append(PMTid, time, charge, timestamp)){
483  Log("ERROR: Appending hits failed!", ERROR, m_verbose);
484  }
485 
486  return sub;
487 }
